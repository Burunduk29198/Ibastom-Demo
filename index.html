Clean PHP templates with ACF integration

// Listing Card Component - template-parts/listing-card.php
<div class="listing-card">
    <div class="listing-image">
        <?php the_post_thumbnail('large', ['style' => 'aspect-ratio: 16/9']); ?>
        <div class="listing-price" dir="ltr">
            ₪<?php echo number_format(get_field('re_listing_price')); ?>
        </div>
    </div>
    <div class="listing-content">
        <h3><?php the_title(); ?></h3>
        <div class="listing-specs">
            <span dir="ltr"><?php echo get_field('re_property_size_sqm'); ?> מ"ר</span>
        </div>
    </div>
</div>

// ACF Flexible Content Loop - single-product.php
<?php if (have_rows('re_content_sections')): ?>
    <?php while (have_rows('re_content_sections')): the_row(); ?>
        <?php if (get_row_layout() == 'features_table'): ?>
            <?php get_template_part('template-parts/sections/features-table'); ?>
        <?php elseif (get_row_layout() == 'pricing_block'): ?>
            <?php get_template_part('template-parts/sections/pricing-block'); ?>
        <?php endif; ?>
    <?php endwhile; ?>
<?php endif; ?>
